function claude-list {

function Test-ClaudeManagedFunctionDefinition {
    param(
        [Parameter(Mandatory = $true)]
        [System.Management.Automation.Language.FunctionDefinitionAst]$FunctionAst,
        [Parameter(Mandatory = $true)]
        [string]$Name
    )

    if ($FunctionAst.Name -ne "claude-$Name") {
        return $false
    }

    $statements = $FunctionAst.Body.EndBlock.Statements
    if ($statements.Count -ne 1) {
        return $false
    }

    $pipeline = $statements[0]
    if ($pipeline -isnot [System.Management.Automation.Language.PipelineAst]) {
        return $false
    }

    $command = $pipeline.PipelineElements[0]
    if ($command -isnot [System.Management.Automation.Language.CommandAst]) {
        return $false
    }

    $elements = $command.CommandElements
    return $command.GetCommandName() -eq "claude" `
        -and $elements.Count -eq 4 `
        -and $elements[1].Extent.Text -eq "--settings" `
        -and $elements[2].Extent.Text -eq "`$HOME/.claude/settings-$Name.json" `
        -and $elements[3].Extent.Text -eq "@args"
}

function Get-LineStartOffset {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Text,
        [Parameter(Mandatory = $true)]
        [int]$Offset
    )

    while ($Offset -gt 0 -and $Text[$Offset - 1] -notin "`r", "`n") {
        $Offset--
    }

    return $Offset
}

function Get-LineEndOffset {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Text,
        [Parameter(Mandatory = $true)]
        [int]$Offset
    )

    while ($Offset -lt $Text.Length -and $Text[$Offset] -notin "`r", "`n") {
        $Offset++
    }

    if ($Offset -lt $Text.Length -and $Text[$Offset] -eq "`r") {
        $Offset++
    }

    if ($Offset -lt $Text.Length -and $Text[$Offset] -eq "`n") {
        $Offset++
    }

    return $Offset
}

function Remove-ClaudeManagedFunctionFromProfileContent {
    param(
        [AllowEmptyString()]
        [string]$ProfileContent,
        [Parameter(Mandatory = $true)]
        [string]$Name
    )

    if ($null -eq $ProfileContent) {
        $ProfileContent = ""
    }

    $tokens = $null
    $errors = $null
    $ast = [System.Management.Automation.Language.Parser]::ParseInput(
        $ProfileContent,
        [ref]$tokens,
        [ref]$errors
    )
    if (@($errors).Count -gt 0) {
        throw "profile 解析失败：$($errors[0].Message)"
    }

    $targets = $ast.FindAll(
        {
            param($node)
            $node -is [System.Management.Automation.Language.FunctionDefinitionAst] `
                -and (Test-ClaudeManagedFunctionDefinition -FunctionAst $node -Name $Name)
        },
        $true
    ) | Sort-Object { $_.Extent.StartOffset }

    if (@($targets).Count -eq 0) {
        return [pscustomobject]@{
            Removed = $false
            Content = $ProfileContent
        }
    }

    $builder = [System.Text.StringBuilder]::new()
    $cursor = 0

    foreach ($target in $targets) {
        $start = Get-LineStartOffset -Text $ProfileContent -Offset $target.Extent.StartOffset
        $end = Get-LineEndOffset -Text $ProfileContent -Offset $target.Extent.EndOffset
        [void]$builder.Append($ProfileContent.Substring($cursor, $start - $cursor))
        $cursor = $end
    }

    [void]$builder.Append($ProfileContent.Substring($cursor))

    return [pscustomobject]@{
        Removed = $true
        Content = $builder.ToString()
    }
}

function New-ClaudeManagedFunctionBlock {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Name
    )

    return @"

function claude-$Name {
    claude --settings `$HOME/.claude/settings-$Name.json @args
}
"@
}

function Get-ClaudeEnvFieldValue {
    param(
        [Parameter(Mandatory = $true)]
        [AllowNull()]
        [object]$EnvConfig,
        [Parameter(Mandatory = $true)]
        [string]$Key
    )

    if ($null -eq $EnvConfig) {
        return $null
    }

    $property = $EnvConfig.PSObject.Properties[$Key]
    if ($null -eq $property) {
        return $null
    }

    return [string]$property.Value
}

function Read-ClaudeUpdateFieldValue {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Label,
        [AllowNull()]
        [string]$CurrentValue,
        [Parameter(Mandatory = $true)]
        [string]$Prompt,
        [switch]$PreserveOnBlank
    )

    $displayValue = if ([string]::IsNullOrWhiteSpace($CurrentValue)) {
        "(未设置)"
    }
    else {
        $CurrentValue
    }

    Write-Host "当前 $Label：$displayValue" -ForegroundColor DarkGray
    $inputValue = (Read-Host $Prompt).Trim()

    if ([string]::IsNullOrWhiteSpace($inputValue)) {
        if ($PreserveOnBlank) {
            return $CurrentValue
        }

        return $null
    }

    return $inputValue
}

function Merge-ClaudeUpdatedEnvBlock {
    param(
        [AllowNull()]
        [object]$EnvConfig,
        [Parameter(Mandatory = $true)]
        [hashtable]$UpdatedValues,
        [Parameter(Mandatory = $true)]
        [string[]]$ManagedKeys
    )

    $envBlock = [ordered]@{}

    if ($null -ne $EnvConfig) {
        foreach ($property in $EnvConfig.PSObject.Properties) {
            $key = $property.Name
            $value = if ($UpdatedValues.ContainsKey($key)) {
                $UpdatedValues[$key]
            }
            else {
                $property.Value
            }

            if ($null -ne $value) {
                $envBlock[$key] = $value
            }
        }
    }

    foreach ($key in $ManagedKeys) {
        if ($UpdatedValues.ContainsKey($key) -and -not $envBlock.Contains($key)) {
            $value = $UpdatedValues[$key]
            if ($null -ne $value) {
                $envBlock[$key] = $value
            }
        }
    }

    return $envBlock
}

function claude-add {
    Write-Host ""
    Write-Host "=== Claude 中转站配置向导 ===" -ForegroundColor Cyan                                                        Write-Host ""
    # ── 1. 名称 ──────────────────────────────────────────
    $name = (Read-Host "中转站名称 ").Trim().ToLower()
    if ([string]::IsNullOrWhiteSpace($name)) { Write-Error "名称不能为空"; return }

    $settingsFile = "$HOME\.claude\settings-$name.json"
    if (Test-Path $settingsFile) {
        $ow = Read-Host "配置 '$name' 已存在，覆盖？[y/N]"
        if ($ow -notmatch '^[yY]$') { return }
    }

    # ── 2. Base URL（留空不写入）─────────────────────────
    $baseUrl = (Read-Host "Base URL (留空 = 不写入)").Trim()

    # ── 3. 认证方式 ───────────────────────────────────────
    Write-Host ""
    Write-Host "认证方式：" -ForegroundColor Yellow
    Write-Host "  1) ANTHROPIC_API_KEY   [默认]"
    Write-Host "  2) ANTHROPIC_AUTH_TOKEN"
    $authChoice = (Read-Host "选择 [1/2，回车默认1]").Trim()

    if ($authChoice -eq "2") {
        $authKey = "ANTHROPIC_AUTH_TOKEN"
        $authValue = (Read-Host "Auth Token").Trim()
    }
    else {
        $authKey = "ANTHROPIC_API_KEY"
        $authValue = (Read-Host "API Key").Trim()
    }
    if ([string]::IsNullOrWhiteSpace($authValue)) { Write-Error "认证凭据不能为空"; return }

    # ── 4. 可选模型环境变量（留空用默认值）──────────────
    Write-Host ""
    Write-Host "模型参数（留空使用默认值）：" -ForegroundColor Yellow
    $inputOpus = (Read-Host "ANTHROPIC_DEFAULT_OPUS_MODEL   [默认 claude-opus-4-6]").Trim()
    $inputSonnet = (Read-Host "ANTHROPIC_DEFAULT_SONNET_MODEL [默认 claude-sonnet-4-6]").Trim()
    $inputModel = (Read-Host "ANTHROPIC_MODEL                [默认 claude-opus-4-6]").Trim()

    $defaultOpus = if ($inputOpus) { $inputOpus }   else { "claude-opus-4-6" }
    $defaultSonnet = if ($inputSonnet) { $inputSonnet } else { "claude-sonnet-4-6" }
    $anthropicModel = if ($inputModel) { $inputModel }  else { "claude-opus-4-6" }

    # ── 5. model 顶层字段（留空用默认值）────────────────
    Write-Host ""
    Write-Host "model 顶层字段：" -ForegroundColor Yellow
    Write-Host "  1) opus[1m]            [默认]"
    Write-Host "  2) claude-sonnet-4-6"
    Write-Host "  3) claude-opus-4-6"
    Write-Host "  4) claude-haiku-4-5-20251001"
    Write-Host "  5) 自定义"
    $mc = (Read-Host "选择 [1-5，回车默认1]").Trim()
    $model = switch ($mc) {
        "2" { "claude-sonnet-4-6" }
        "3" { "claude-opus-4-6" }
        "4" { "claude-haiku-4-5-20251001" }
        "5" { (Read-Host "输入 model 值").Trim() }
        default { "opus[1m]" }
    }

    # ── 6. 组装 env 块 ────────────────────────────────────
    $envBlock = [ordered]@{}
    if ($baseUrl) { $envBlock["ANTHROPIC_BASE_URL"] = $baseUrl }
    $envBlock[$authKey] = $authValue
    $envBlock["ANTHROPIC_DEFAULT_OPUS_MODEL"] = $defaultOpus
    $envBlock["ANTHROPIC_DEFAULT_SONNET_MODEL"] = $defaultSonnet
    $envBlock["ANTHROPIC_MODEL"] = $anthropicModel

    # ── 7. 组装完整配置对象 ───────────────────────────────
    $config = [ordered]@{
        "model" = $model
        "env"   = $envBlock
    }

    # ── 8. 创建目录并保存配置 ─────────────────────────────
    $dir = "$HOME\.claude"
    if (!(Test-Path $dir)) { New-Item -ItemType Directory -Path $dir -Force | Out-Null }

    [System.IO.File]::WriteAllText($settingsFile, ($config | ConvertTo-Json -Depth 5),
        [System.Text.UTF8Encoding]::new($false))
    Write-Host "✓ 配置已保存到: $settingsFile" -ForegroundColor Green
    Write-Host ""

    # ── 9. 追加函数到 profile ─────────────────────────────
    $profilePath = $PROFILE.CurrentUserCurrentHost
    $profileContent = Get-Content $profilePath -Raw -Encoding UTF8 -ErrorAction SilentlyContinue

    if ($profileContent -match "function claude-$name\b") {
        Write-Host "✓ 函数 claude-$name 已在 profile 中，跳过注册" -ForegroundColor Yellow
    }
    else {
        $block = New-ClaudeManagedFunctionBlock -Name $name
        Add-Content -Path $profilePath -Value $block -Encoding UTF8
        Write-Host "✓ 已注册函数 claude-$name 到 profile" -ForegroundColor Green
    }

    # ── 10. 当前会话立即生效 ──────────────────────────────
    $scriptBlock = [scriptblock]::Create("claude --settings `"$settingsFile`" @args")
    New-Item -Path "Function:claude-$name" -Value $scriptBlock -Force | Out-Null

    Write-Host "✓ 完成！当前会话已可直接使用：claude-$name" -ForegroundColor Cyan
    Write-Host ""
}



function claude-delete {
    Write-Host ""
    Write-Host "=== Claude 中转站删除向导 ===" -ForegroundColor Cyan
    Write-Host ""

    # ── 1. 展示现有中转站 ─────────────────────────────────
    $files = Get-ChildItem "$HOME\.claude\settings-*.json" -ErrorAction SilentlyContinue
    if (-not $files) {
        Write-Host "暂无任何中转站配置。" -ForegroundColor Yellow
        return
    }

    Write-Host ("  {0,-4} {1,-16} {2,-20} {3}" -f "序号", "命令", "model 字段", "Base URL")
    Write-Host ("  " + "-" * 68)

    $list = @()
    $i = 1
    foreach ($f in $files) {
        $n = $f.BaseName -replace "^settings-", ""
        try {
            $cfg = Get-Content -Path $f.FullName -Raw -Encoding UTF8 | ConvertFrom-Json
            $model = if ($cfg.model) { $cfg.model } else { "(未指定)" }
            $url = if ($cfg.env.ANTHROPIC_BASE_URL) { $cfg.env.ANTHROPIC_BASE_URL } else { "官方" }
        }
        catch {
            $model = "(解析失败)"; $url = ""
        }
        Write-Host ("  {0,-4} {1,-16} {2,-20} {3}" -f $i, "claude-$n", $model, $url)
        $list += $n
        $i++
    }

    Write-Host ""

    # ── 2. 选择要删除的中转站 ────────────────────────────
    $choice = (Read-Host "输入序号或名称删除 (回车取消)").Trim()
    if ([string]::IsNullOrWhiteSpace($choice)) {
        Write-Host "已取消。" -ForegroundColor DarkGray
        return
    }

    $targetName = ""
    if ($choice -match '^\d+$') {
        $idx = [int]$choice - 1
        if ($idx -lt 0 -or $idx -ge $list.Count) {
            Write-Error "序号超出范围"; return
        }
        $targetName = $list[$idx]
    }
    else {
        $targetName = $choice.ToLower()
        if ($list -notcontains $targetName) {
            Write-Error "找不到中转站：$targetName"; return
        }
    }

    # ── 3. 确认删除 ───────────────────────────────────────
    Write-Host ""
    $confirm = Read-Host "确认删除 claude-$targetName ？[y/N]"
    if ($confirm -notmatch '^[yY]$') {
        Write-Host "已取消。" -ForegroundColor DarkGray
        return
    }

    # ── 4. 删除 settings JSON ─────────────────────────────
    $settingsFile = "$HOME\.claude\settings-$targetName.json"
    if (Test-Path $settingsFile) {
        Remove-Item $settingsFile -Force
        Write-Host "✓ 已删除：$settingsFile" -ForegroundColor Green
    }

    # ── 5. 从 profile 中移除函数块 ───────────────────────
    $profilePath = $PROFILE.CurrentUserCurrentHost
    $profileContent = Get-Content $profilePath -Raw -Encoding UTF8 -ErrorAction SilentlyContinue
    $removeResult = Remove-ClaudeManagedFunctionFromProfileContent `
        -ProfileContent $profileContent `
        -Name $targetName

    if ($removeResult.Removed) {
        $newContent = $removeResult.Content
        [System.IO.File]::WriteAllText($profilePath, $newContent, [System.Text.UTF8Encoding]::new($false))
        Write-Host "✓ 已从 profile 中移除函数 claude-$targetName" -ForegroundColor Green
    }
    else {
        Write-Host "  profile 中未找到函数 claude-$targetName，跳过" -ForegroundColor DarkGray
    }

    # ── 6. 当前会话移除函数 ───────────────────────────────
    if (Get-Item -Path "Function:claude-$targetName" -ErrorAction SilentlyContinue) {
        Remove-Item -Path "Function:claude-$targetName" -Force
        Write-Host "✓ 当前会话已移除函数 claude-$targetName" -ForegroundColor Green
    }

    Write-Host ""
    Write-Host "✓ 完成！claude-$targetName 已彻底删除。" -ForegroundColor Cyan
    Write-Host ""
}

function claude-update {
    Write-Host ""
    Write-Host "=== Claude 中转站更新向导 ===" -ForegroundColor Cyan
    Write-Host ""

    # ── 1. 展示现有中转站 ─────────────────────────────────
    $files = Get-ChildItem "$HOME\.claude\settings-*.json" -ErrorAction SilentlyContinue
    if (-not $files) {
        Write-Host "暂无任何中转站配置。" -ForegroundColor Yellow
        return
    }

    Write-Host ("  {0,-4} {1,-16} {2,-20} {3}" -f "序号", "命令", "model 字段", "Base URL")
    Write-Host ("  " + "-" * 68)

    $list = @()
    $i = 1
    foreach ($f in $files) {
        $n = $f.BaseName -replace "^settings-", ""
        try {
            $cfg = Get-Content -Path $f.FullName -Raw -Encoding UTF8 | ConvertFrom-Json
            $model = if ($cfg.model) { $cfg.model } else { "(未指定)" }
            $url = if ($cfg.env.ANTHROPIC_BASE_URL) { $cfg.env.ANTHROPIC_BASE_URL } else { "官方" }
        }
        catch {
            $model = "(解析失败)"; $url = ""
        }
        Write-Host ("  {0,-4} {1,-16} {2,-20} {3}" -f $i, "claude-$n", $model, $url)
        $list += $n
        $i++
    }

    Write-Host ""

    # ── 2. 选择目标中转站 ────────────────────────────────
    $choice = (Read-Host "输入序号或名称 (回车取消)").Trim()
    if ([string]::IsNullOrWhiteSpace($choice)) {
        Write-Host "已取消。" -ForegroundColor DarkGray
        return
    }

    $targetName = ""
    if ($choice -match '^\d+$') {
        $idx = [int]$choice - 1
        if ($idx -lt 0 -or $idx -ge $list.Count) {
            Write-Error "序号超出范围"; return
        }
        $targetName = $list[$idx]
    }
    else {
        $targetName = $choice.ToLower()
        if ($list -notcontains $targetName) {
            Write-Error "找不到中转站：$targetName"; return
        }
    }

    # ── 3. 读取当前配置 ───────────────────────────────────
    $settingsFile = "$HOME\.claude\settings-$targetName.json"
    $cfg = Get-Content -Path $settingsFile -Raw -Encoding UTF8 | ConvertFrom-Json

    $currentUrl = Get-ClaudeEnvFieldValue -EnvConfig $cfg.env -Key "ANTHROPIC_BASE_URL"
    $currentDefaultOpusModel = Get-ClaudeEnvFieldValue -EnvConfig $cfg.env -Key "ANTHROPIC_DEFAULT_OPUS_MODEL"
    $currentDefaultSonnetModel = Get-ClaudeEnvFieldValue -EnvConfig $cfg.env -Key "ANTHROPIC_DEFAULT_SONNET_MODEL"
    $currentAnthropicModel = Get-ClaudeEnvFieldValue -EnvConfig $cfg.env -Key "ANTHROPIC_MODEL"

    Write-Host ""
    $newUrl = Read-ClaudeUpdateFieldValue `
        -Label "Base URL" `
        -CurrentValue $currentUrl `
        -Prompt "新 Base URL (留空 = 清除该字段)"
    Write-Host ""
    $newDefaultOpusModel = Read-ClaudeUpdateFieldValue `
        -Label "ANTHROPIC_DEFAULT_OPUS_MODEL" `
        -CurrentValue $currentDefaultOpusModel `
        -Prompt "新 ANTHROPIC_DEFAULT_OPUS_MODEL (留空 = 保留原值)" `
        -PreserveOnBlank
    Write-Host ""
    $newDefaultSonnetModel = Read-ClaudeUpdateFieldValue `
        -Label "ANTHROPIC_DEFAULT_SONNET_MODEL" `
        -CurrentValue $currentDefaultSonnetModel `
        -Prompt "新 ANTHROPIC_DEFAULT_SONNET_MODEL (留空 = 保留原值)" `
        -PreserveOnBlank
    Write-Host ""
    $newAnthropicModel = Read-ClaudeUpdateFieldValue `
        -Label "ANTHROPIC_MODEL" `
        -CurrentValue $currentAnthropicModel `
        -Prompt "新 ANTHROPIC_MODEL (留空 = 保留原值)" `
        -PreserveOnBlank

    # ── 5. 修改并写回 JSON ────────────────────────────────
    $managedKeys = @(
        "ANTHROPIC_BASE_URL",
        "ANTHROPIC_DEFAULT_OPUS_MODEL",
        "ANTHROPIC_DEFAULT_SONNET_MODEL",
        "ANTHROPIC_MODEL"
    )
    $updatedValues = @{
        ANTHROPIC_BASE_URL             = $newUrl
        ANTHROPIC_DEFAULT_OPUS_MODEL   = $newDefaultOpusModel
        ANTHROPIC_DEFAULT_SONNET_MODEL = $newDefaultSonnetModel
        ANTHROPIC_MODEL                = $newAnthropicModel
    }
    $envBlock = Merge-ClaudeUpdatedEnvBlock `
        -EnvConfig $cfg.env `
        -UpdatedValues $updatedValues `
        -ManagedKeys $managedKeys

    $newConfig = [ordered]@{
        model = $cfg.model
        env   = $envBlock
    }

    [System.IO.File]::WriteAllText(
        $settingsFile,
        ($newConfig | ConvertTo-Json -Depth 5),
        [System.Text.UTF8Encoding]::new($false)
    )

    Write-Host ""
    if ($newUrl) {
        Write-Host "✓ Base URL 已更新为：$newUrl" -ForegroundColor Green
    }
    else {
        Write-Host "✓ Base URL 已清除（将使用官方端点）" -ForegroundColor Green
    }
    Write-Host "✓ 已保存：$settingsFile" -ForegroundColor Green
    Write-Host ""
}

function claude-sync {
    Write-Host ""                                                                                                           Write-Host "=== Claude 中转站同步检查 ===" -ForegroundColor Cyan
    Write-Host ""

    $profilePath = $PROFILE.CurrentUserCurrentHost
    $profileContent = Get-Content $profilePath -Raw -Encoding UTF8 -ErrorAction SilentlyContinue

    $files = Get-ChildItem "$HOME\.claude\settings-*.json" -ErrorAction SilentlyContinue
    if (-not $files) {
        Write-Host "~/.claude 中无任何中转站配置文件。" -ForegroundColor Yellow
        return
    }

    $added = @()
    $skipped = @()

    foreach ($f in $files) {
        $name = $f.BaseName -replace "^settings-", ""

        # 跳过官方通用 settings.json（BaseName 为 "settings"，不含 -xx，已被过滤）
        # 额外跳过 settings 本身（以防万一）
        if ($name -eq "settings") { continue }

        if ($profileContent -match "function claude-$name\b") {
            $skipped += "claude-$name"
        }
        else {
            # 追加到 profile
            $block = New-ClaudeManagedFunctionBlock -Name $name
            Add-Content -Path $profilePath -Value $block -Encoding UTF8

            # 当前会话立即生效
            $settingsFile = "$HOME\.claude\settings-$name.json"
            $scriptBlock = [scriptblock]::Create("claude --settings `"$settingsFile`" @args")
            New-Item -Path "Function:claude-$name" -Value $scriptBlock -Force | Out-Null

            $added += "claude-$name"
        }
    }

    # ── 输出结果 ──────────────────────────────────────────
    if ($skipped) {
        Write-Host "已存在（跳过）：" -ForegroundColor DarkGray
        foreach ($n in $skipped) {
            Write-Host "    $n" -ForegroundColor DarkGray
        }
        Write-Host ""
    }

    if ($added) {
        Write-Host "已注册（新增）：" -ForegroundColor Green
        foreach ($n in $added) {
            Write-Host "    $n" -ForegroundColor Green
        }
        Write-Host ""
        Write-Host "✓ 同步完成，当前会话已可直接使用上述命令。" -ForegroundColor Cyan
    }
    else {
        Write-Host "✓ 所有中转站均已注册，无需同步。" -ForegroundColor Cyan
    }

    Write-Host ""
}
